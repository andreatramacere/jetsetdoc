    #print nu_IC_blob,target_nu,target_nu*nu_IC_blob

   
    jet.eval()
    
    
    nu_obs=sp.array([])

    nuFnu_target = sp.power(10.0,sp.interp(sp.log10(target_nu_obs),nu_obs,nuFnu_obs))

        target_nu_obs=sp.array([target_nu_obs])

    jet.set_flag(flag_initial)
